    function renderProfile(data) {
        document.getElementById('api-name').innerText = data.name;
        document.getElementById('api-sign').innerText = data.signature;
        document.getElementById('api-avatar').src = data.avatar;
        document.getElementById('api-cover').src = data.cover;
        const momentBox = document.getElementById('api-moments');
        momentBox.innerHTML = data.moments.map(img => `<img class="m-img" src="${img}">`).join('');
        const archiveBox = document.getElementById('api-archives');
        archiveBox.innerHTML = data.archives.map(item => `
            <div class="archive-item">
                <div class="archive-label">${item.label}</div>
                <div class="archive-val">${item.val}</div>
            </div>
        `).join('');
        const footBox = document.getElementById('api-footprints');
        footBox.innerHTML = data.footprints.map(node => `
            <div class="foot-node">
                <div class="foot-text">${node.text}</div>
                <div class="foot-time">${node.time}</div>
            </div>
        `).join('');
    }
    /* 菜单控制 */
    function toggleSheet(show) {
        document.body.classList.toggle('sheet-active', show);
    }
    /* --- 新增：中央对话框控制逻辑 --- */
    function showCustomModal(config) {
        const mask = document.getElementById('customModal');
        const content = document.getElementById('modalContent');
        content.innerHTML = `
            <div class="modal-body">
                <div class="modal-title">${config.title}</div>
                <div class="modal-text">${config.text}</div>
                ${config.showInput ? `<input type="text" id="mInput" class="modal-input" placeholder="${config.placeholder}" value="${config.inputValue || ''}">` : ''}
            </div>
            <div class="modal-footer">
                <button class="modal-btn cancel" onclick="closeCustomModal()">取消</button>
                <button class="modal-btn ${config.isDanger ? 'danger' : 'confirm'}" id="mConfirm">确定</button>
            </div>
        `;
        mask.style.display = 'flex';
        setTimeout(() => mask.classList.add('modal-show'), 10);
        document.getElementById('mConfirm').onclick = () => {
            const inputVal = config.showInput ? document.getElementById('mInput').value : null;
            config.onConfirm(inputVal);
            closeCustomModal();
        };
    }
    function closeCustomModal() {
        const mask = document.getElementById('customModal');
        mask.classList.remove('modal-show');
        setTimeout(() => mask.style.display = 'none', 200);
    }
    /* 替换后的点击功能 */
    function modifyNote() {
        toggleSheet(false);
        showCustomModal({
            title: "修改备注名",
            text: "请为你的伙伴设置一个专属称呼",
            showInput: true,
            placeholder: "输入备注...",
            inputValue: document.getElementById('api-name').innerText,
            onConfirm: (val) => {
                if(val) document.getElementById('api-name').innerText = val;
            }
        });
    }
    function confirmBlock() {
        toggleSheet(false);
        showCustomModal({
            title: "确定拉黑？",
            text: "拉黑后将不再接收对方的消息通知。",
            isDanger: true,
            onConfirm: () => {
                showCustomModal({ title: "提示", text: "已成功拉黑该伙伴", onConfirm: () => {} });
            }
        });
    }
    function confirmDelete() {
        toggleSheet(false);
        showCustomModal({
            title: "解除同行关系",
            text: "确定要解除吗？此操作将清空所有共同记忆。",
            isDanger: true,
            onConfirm: () => {
                history.back();
            }
        });
